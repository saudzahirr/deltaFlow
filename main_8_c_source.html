34int main(int argc, char* argv[]) {
35 ArgumentParser opts(argc, argv);
36 DEBUG("Bus Data csv :: {}", opts.getBusDataCsv());
37 DEBUG("Branch Data csv :: {}", opts.getBranchDataCsv());
38
39 BusData busData;
40 BranchData branchData;
41
42 Solver solver = opts.getSolver();
43 int maxIter = opts.getMaxIterations();
44 double tolerance = opts.getTolerance();
45
46 readBusDataCSV(opts.getBusDataCsv(), busData);
47 readBranchDataCSV(opts.getBranchDataCsv(), branchData);
48
49 auto Y = computeAdmittanceMatrix(busData, branchData);
50
51 switch (solver) {
52 case Solver::GaussSeidel: {
53 double relaxation_coeff = opts.getRelaxationCoefficient();
54 GaussSeidel(Y, busData, maxIter, tolerance, relaxation_coeff);
55 break;
56 }
57
58 case Solver::NewtonRaphson: {
59 NewtonRaphson(Y, busData, maxIter, tolerance);
60 break;
61 }
62
63 default: {
64 NewtonRaphson(Y, busData, maxIter, tolerance);
65 break;
66 }
67 }
68
69 dispBusData(busData);
70 dispLineFlow(busData, branchData, Y);
71
72 writeOutputCSV(busData);
73
74 return 0;
75}
