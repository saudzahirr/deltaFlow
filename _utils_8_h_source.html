70 inline void disp(const Eigen::MatrixXcd& Y, int width = 22, bool left_justify = false) {
71 for (int i = 0; i < Y.rows(); ++i) {
72 std::stringstream rowStream;
73 for (int j = 0; j < Y.cols(); ++j) {
74 auto R = Y(i, j).real();
75 auto X = Y(i, j).imag();
76
77 std::stringstream cell;
78 if (X >= 0) {
79 cell << "(" << R << " + " << X << "i)";
80 } else {
81 cell << "(" << R << " - " << -X << "i)";
82 }
83
84 if (left_justify)
85 rowStream << std::left;
86 else
87 rowStream << std::right;
88
89 rowStream << std::setw(width) << cell.str();
90
91 if (j != Y.cols() - 1)
92 rowStream << "\t";
93 }
94 DEBUG("{}", rowStream.str().c_str());
95 }
96 }
