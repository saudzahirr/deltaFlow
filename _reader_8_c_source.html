167 }

178 INFO("{}", std::string(80, '='));

180 for (int n = 1; n <= nBus; ++n) {
181 int n_idx = n - 1;
182 bool busprt = false;
183
184 for (int L = 0; L < nLine; ++L) {
185 if (!busprt) {

187 double Q_inj = busData.Qg(n_idx) - busData.Ql(n_idx);
188 double S_mag = std::abs(S(n_idx)) * basemva;

190 INFO("{:6d} {:9.3f} {:9.3f} {:9.3f}", n, P_inj, Q_inj, S_mag);
191 busprt = true;
192 }
193
194 if (branchData.From(L) == n) {

198

229 int k_idx = k - 1;

231

233 std::complex<double> Ik = (V(k_idx) - aL * V(n_idx)) * Y(L) / (aL * aL) + Bc(L) / (aL * aL) * V(k_idx);
